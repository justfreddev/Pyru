// Prints out nth fibonacci number up to n=20
def fib(n) {
  if (n <= 1) return n;
  return fib(n - 2) + fib(n - 1);
}
for (var i = 0; i < 20; i++) {
  print fib(i);
}


// Prints out 5, 4, 3, 2, 1, 0
def recursion(x) {
  print x;
  if (x == 0) return x;
  return recursion(x - 1);
}
recursion(5);

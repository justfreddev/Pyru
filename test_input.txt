fun fib(n) {
  if (n <= 1) return n;
  return fib(n - 2) + fib(n - 1);
}
for (var i = 0; i < 20; i++) {
  print fib(i);
}

fun recursion(x) {
  if (x == 0) return x;
  recursion(x - 1);
}
